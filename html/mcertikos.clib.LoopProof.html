Section S.

  Context `{Hcc: Events.CompilerConfiguration}
          `{Hwb: Events.WritableBlock}.

  Variables (body: statement) (genv: genv) (env: env)
            (P Q: temp_env → mem → Prop)
            (R: temp_env→ mem → option (val × type)-> Prop).

  Record t : Type := make {
         W: Type;
         lt: W→ W→ Prop;
         lt_wf: well_founded lt;
         I: temp_env→ mem→ W→ Prop;
         P_implies_I: ∀ le m, P le m→ ∃ n0, I le m n0;
         I_invariant:
         ∀ le m n, I le m n →
           ∃ out le´ m´, exec_stmt genv env le m body E0 le´ m´ out ∧
             ((out = Out_normal ∨ out = Out_continue)-> ∃ n´, lt n´ n ∧ I le´ m´ n´) ∧
              (out = Out_break→ Q le´ m´) ∧
              (∀ v, out = Out_return v→ R le´ m´ v)
  }.

  Theorem termination: t → ∀ le m, P le m →
    (∃ out le´ m´, exec_stmt genv env le m (Sloop body Sskip) E0 le´ m´ out ∧
      ((∃ v, out = Out_return v ∧ R le´ m´ v) ∨ (out = Out_normal ∧ Q le´ m´))).
  Proof.
    destruct 1; simpl.
    intros.
    generalize (P_implies_I0 le m H).
    intro.
    destruct H0 as [n0].
    clear H.
    revert le m H0.
    induction n0 using (well_founded_ind lt_wf0).

    intros.

    generalize (I_invariant0 le m n0 H0).
    intro I_invariant.
    destruct I_invariant as [out tinv].
    destruct tinv as [le´ tinv].
    destruct tinv as [m´ tinv].
    destruct tinv as [Inv1 tinv].
    destruct tinv as [Inv2 tinv].
    destruct tinv as [Inv3 Inv4].

    destruct out.

    ∃ Out_normal.
    ∃ le´.
    ∃ m´.
    split.
    econstructor.
    eassumption.
    econstructor.
    right.
    split.
    trivial.
    apply Inv3.
    trivial.

    destruct Inv2 as [n´].
    auto.
    destruct H1.
    generalize (H n´ H1 le´ m´ H2).
    intro.
    destruct H3 as [out].
    destruct H3 as [le´0´].
    destruct H3 as [m´0].
    destruct H3.
    destruct H4.
      destruct H4 as [v].
      destruct H4.
      esplit. esplit. esplit.

      eassumption.
      econstructor.
      econstructor.
      eassumption.

Module LoopProofWhileWithContinue.

Section S.

  Context `{Hcc: Events.CompilerConfiguration}
          `{Hwb: Events.WritableBlock}.

  Variables (condition: expr) (body: statement)
            (genv: genv) (env: env)
            (P Q: temp_env → mem → Prop).

  Record t : Type := make {
         W: Type;
         lt: W→ W→ Prop;

         P_implies_I: ∀ le m, P le m→ ∃ n0, I le m n0;
         I_invariant:
         ∀ le m n, I le m n→ ∃ v b, (eval_expr genv env le m condition v ∧
            (bool_val v (typeof condition) = Some b) ∧
            (b = false → Q le m) ∧
            (b = true →
             ∃ out le´ m´, exec_stmt genv env le m body E0 le´ m´ out ∧
             (out = Out_normal ∨ out = Out_continue) ∧
             ∃ n´, lt n´ n ∧ I le´ m´ n´
           ))
  }.

  Theorem termination: t → ∀ le m, P le m →
    (∃ le´ m´, exec_stmt genv env le m (Swhile condition body) E0 le´ m´ Out_normal ∧ Q le´ m´).

    destruct Inv2.

    subst.

    destruct H1 as [m´0].
    destruct H1.
