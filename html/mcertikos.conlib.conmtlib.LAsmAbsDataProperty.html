  Lemma decode_longs_ll_lessdef vs1 vs2 x1 y1:
    x1 :: y1 :: nil = decode_longs (sig_args SelectLong.sig_ll_l) vs1 →
    Val.lessdef_list vs1 vs2 →
    ∃ x2 y2,
      x2 :: y2 :: nil = decode_longs (sig_args SelectLong.sig_ll_l) vs2 ∧
      Val.lessdef x1 x2 ∧
      Val.lessdef y1 y2.
  Proof.
