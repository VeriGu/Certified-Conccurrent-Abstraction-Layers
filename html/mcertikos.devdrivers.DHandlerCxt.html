      unfold ioapic_trans_intr in *;
      subdestruct; eauto.

      eapply nth_error_valid_length in Hdestruct1; eauto.
      generalize valid_ioapic_enables, Hdestruct1.
      repeat match goal with | [H: _ |- _] ⇒ clear H end.

      change (Z.of_nat 257%nat) with 257.

    Qed.
