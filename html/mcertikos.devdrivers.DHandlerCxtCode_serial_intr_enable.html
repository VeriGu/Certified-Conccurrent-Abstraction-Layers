        Function serial_intr_enable_concrete_rec (n: nat) (abd: RData) {struct n} : option (RData × bool) :=

          | O ⇒ Some (abd, true)
          | S n´ ⇒
            match serial_intr_enable_concrete_rec n´ abd with
            | Some (d´, true) ⇒ match serial_intr_pending_handler d´ with
                                | Some (d´´, true) ⇒ Some (d´´, true)
                                | Some (d´´, false) ⇒ Some (d´, false)
                                | None ⇒ None
                                end
            | Some (d´, false) ⇒ Some (d´, false)
            | None ⇒ None
            end
          end.
