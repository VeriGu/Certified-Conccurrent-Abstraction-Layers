Lemma serial_calc_com1_base:
  ∀ d,
    high_level_invariant d →
    Base (s (com1 d)) = 1016.
Proof.
  intros d Hinv.
  generalize (valid_com1_port d); intros Hport; apply Hport in Hinv.
  rewrite Hinv. reflexivity.
Qed.
