          ∀ m d le t h tl c,

            SerialEnv ((l2 (com1 d)) + t) = SerialSendComplete →
            (∀ n, 0 ≤ n < t → SerialEnv ((l2 (com1 d)) + n) = NullEvent) →
            high_level_invariant d →

            RxBuf (s (com1 d)) = nil →
            TxBuf (s (com1 d)) = h :: tl →
