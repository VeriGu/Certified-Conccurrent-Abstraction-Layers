  Context `{real_params: RealParams}.

  Inductive get_serial_exists_spec_low_step `{StencilOps} `{Mem.MemoryModelOps} `{UseMemWithData mem}:
    sextcall_sem (mem := mwd LDATAOps) :=
    get_serial_exists_spec_low_intro s (WB: _ → Prop) (m´0: mwd LDATAOps) b0 v:

  Inductive set_serial_exists_spec_low_step `{StencilOps} `{Mem.MemoryModelOps} `{UseMemWithData mem}:
    sextcall_sem (mem := mwd LDATAOps) :=
    | set_serial_exists_spec_low_intro_1 s (WB: _ → Prop) (m´0 m0: mwd LDATAOps) b0 v:
      find_symbol s serial_exists_LOC = Some b0 →
      0 < Int.unsigned v →

      kernel_mode (snd m´0) →

    | set_serial_exists_spec_low_intro_0 s (WB: _ → Prop) (m´0 m0: mwd LDATAOps) b0 v:
      find_symbol s serial_exists_LOC = Some b0 →
