      Definition exec_host_load2high (adr: int) (chunk: memory_chunk) (m: mwd HDATAOps) (rs: regset) (rd: preg) :=
        let adt:= (snd m) in
        let ofs := (Int.unsigned adr) in
        if (zeq (ZMap.get (CPU_ID adt) (cid adt)) (PT adt)) then
          match ZMap.get (PDX ofs) (ZMap.get (PT adt) (ptpool adt))with
          | PDEValid _ pte ⇒
            if (zle (ofs mod PgSize) (PgSize - size_chunk chunk)) then
              if (Zdivide_dec (align_chunk chunk) (ofs mod PgSize) (align_chunk_pos chunk)) then
                match ZMap.get (PTX ofs) pte with
                | PTEValid v p ⇒
                  match p with
                  | PTK _ ⇒ Stuck
                  | _ ⇒ match ZMap.get (PageI (PTADDR v ofs)) (pperm adt) with
                           | PGAlloc ⇒ exec_flatmem_load chunk m (PTADDR v ofs) rs rd
                           | _ ⇒ Stuck
                         end
                  end
                | PTEUnPresent ⇒ exec_pagefault ge m adr rs
                |_ ⇒ Stuck
                end
              else Stuck
            else Stuck
          | _ ⇒ Stuck
          end
        else Stuck.

      Lemma exec_host_load2high_high_level_invariant:
        ∀ adr chunk m rs rd rs´ m´,
          exec_host_load2high adr chunk m rs rd = Next rs´ m´ →
          high_level_invariant (snd m) →
          high_level_invariant (snd m´).
      Proof.
        unfold exec_host_load2high. intros until m´.
        destruct (zeq (ZMap.get (CPU_ID (snd m)) (cid (snd m))) (PT (snd m))); try discriminate.
        destruct (
            ZMap.get (PDX (Int.unsigned adr))
                     (ZMap.get (PT (snd m)) (ptpool (snd m)))
          ); try discriminate.
        destruct (zle (Int.unsigned adr mod 4096) (4096 - size_chunk chunk)); contra_inv.
        destruct (Zdivide_dec (align_chunk chunk) (Int.unsigned adr mod 4096)
                              (align_chunk_pos chunk)); contra_inv.
        destruct (ZMap.get (PTX (Int.unsigned adr)) pte); try discriminate.
        destruct p; try discriminate.
        × destruct (ZMap.get (PageI (PTADDR v (Int.unsigned adr))) (pperm (snd m)));
          try discriminate.
          intros. exploit exec_flatmem_load_mem; eauto. congruence.
        × destruct (ZMap.get (PageI (PTADDR v (Int.unsigned adr))) (pperm (snd m)));
          try discriminate.
          intros. exploit exec_flatmem_load_mem; eauto. congruence.
        × eapply exec_pagefault_high_level_invariant; eauto.
      Qed.

    Lemma host_load_correct2high:
      ∀ {F V} (ge1 ge2: Genv.t F V) t m1 m1´ m2 d1 d2 d1´ rs1 rd rs2 rs1´ f i s,
        exec_host_load2high ge1 i t (m1, d1) rs1 rd = Next rs1´ (m1´, d1´)
        → MatchPrimcallStates (one_crel HDATAOps LDATAOps) s f rs1 m1 d1 rs2 m2 d2
        → stencil_matches s ge1
        → stencil_matches s ge2
        → ∃ r´0 m2´ d2´,
             exec_host_load2high ge2 i t (m2, d2) rs2 rd = Next r´0 (m2´, d2´)
             ∧ MatchPrimcallStates (one_crel HDATAOps LDATAOps) s f rs1´ m1´ d1´ r´0 m2´ d2´.
    Proof.
      Local Opaque Z.sub.
      intros. unfold exec_host_load2high in *; simpl in ×.
      pose proof H0 as HM. inv H0. inv match_extcall_states.
      exploit relate_impl_PT_eq; eauto.
      exploit relate_impl_ptpool_eq; eauto. intros.
      exploit relate_impl_cid_eq; eauto; intros.
      exploit relate_impl_CPU_ID_eq; eauto; intros.
      exploit relate_impl_pperm_eq; eauto; intros.
      rewrite <- H3, <- H4, <- H5, <- H6, <- H0. subdestruct.
      - eapply exec_flatmem_load_correct; eauto.
      - eapply exec_flatmem_load_correct; eauto.
      - eapply pagefault_correct; eauto.
    Qed.

    Lemma host_store_correct2high:
      ∀ {F V} (ge1 ge2: Genv.t F V) t m1 m1´ m2 d1 d2 d1´ rs1 rd rs2 rs1´ f i s ds,
        exec_host_store2high (flatmem_store:= hflatmem_store) ge1 i t (m1, d1) rs1 rd ds = Next rs1´ (m1´, d1´)
        → MatchPrimcallStates (one_crel HDATAOps LDATAOps) s f rs1 m1 d1 rs2 m2 d2
        → stencil_matches s ge1
        → stencil_matches s ge2
        → ∃ r´0 m2´ d2´,
             exec_host_store2high (flatmem_store:= lflatmem_store) ge2 i t (m2, d2) rs2 rd ds = Next r´0 (m2´, d2´)
             ∧ MatchPrimcallStates (one_crel HDATAOps LDATAOps) s f rs1´ m1´ d1´ r´0 m2´ d2´.
    Proof.

      exploit relate_impl_PT_eq; eauto.

      exploit relate_impl_cid_eq; eauto; intros.
      exploit relate_impl_CPU_ID_eq; eauto; intros.
      rewrite <- H3, <- H4, <- H5, <- H0. subdestruct.

        apply PTADDR_mod_lt. assumption.
      - eapply exec_flatmem_store_correct; eauto.
        apply PTADDR_mod_lt. assumption.
      - eapply pagefault_correct; eauto.
    Qed.
