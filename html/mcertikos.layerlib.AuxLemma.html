Lemma Int_unsigned_eq:
  ∀ a b,
    Int.unsigned a = Int.unsigned b → a = b.
Proof.
  intros.
  rewrite <- Int.repr_unsigned with a.
  rewrite <- Int.repr_unsigned with b.
  rewrite H. trivial.
Qed.

Lemma Int_add_aux:
  ∀ n1 n2,
    Int.add (Int.repr n1) (Int.repr n2)
    = Int.repr (n1 + n2).
Proof.
  intros.
  unfold Int.add.
  apply Int.eqm_samerepr.
  apply Int.eqm_add;
    apply Int.eqm_unsigned_repr_l, Int.eqm_refl.
Qed.

    ∀ a b c :Z, a × c - b × c = (a - b) × c.
  Proof.
    intros.

  Lemma eqmod_mod_eq´:
    ∀ x y, x mod Int.modulus = y mod Int.modulus → Int.eqmod (Int.modulus) x y.
  Proof.
    intros.
    unfold Int.eqmod.
    assert(HNE: Int.modulus > 0).
    specialize (Int.Z_mod_modulus_range 0).
    intros.
    omega.
    rewrite Zmod_eq in H; trivial.
    rewrite Zmod_eq in H; trivial.

    set (b := y/Int.modulus) in ×.

    assert(HW1: x = a × Int.modulus - b × Int.modulus + y).
    omega.
    rewrite multi_minus in HW1.
    trivial.
  Qed.

  Lemma inject_forward_equal:
    ∀ s id b b´ delta ι
           (match_inject_forward : ∀ (b1 b2 : block) (delta : Z),
                                     ι b1 = Some (b2, delta) →
                                     delta = 0 ∧ (b1 ≤ b2)%positive)
           (match_inject_flat : inject_incr (Mem.flat_inj (genv_next s)) ι),
      find_symbol s id = Some b →
      ι b´ = Some (b, delta) →
      (b, delta) = (b´, 0).
  Proof.
    intros.
    assert (ι b´ = Some (b´, 0%Z)).
    {
      apply match_inject_flat.
      unfold Mem.flat_inj.
      destruct (plt b´ (genv_next s)) as [Hb´ | Hb´].
      reflexivity.
      elim Hb´.
      apply match_inject_forward in H0.
      apply genv_symb_range in H.
      xomega.
    }
    congruence.
  Qed.
