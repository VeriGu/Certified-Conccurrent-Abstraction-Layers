Lemma multi_log_double_update:
  ∀ d ml1 ml2,
    (d {multi_log: ml1}) {multi_log: ml2} = d {multi_log: ml2}.
Proof.
  intros.
  unfold update_multi_log; simpl.
  reflexivity.
Qed.

Lemma ikern_remain :
  ∀ d ml, ikern d = ikern (d {multi_log: ml}).
Proof. intros; simpl; auto. Qed.

Lemma ihost_remain:
  ∀ d ml, ihost d = ihost (d {multi_log: ml}).
Proof. intros; simpl; auto. Qed.
