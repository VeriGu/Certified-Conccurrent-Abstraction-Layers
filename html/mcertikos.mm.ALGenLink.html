    ∀ COMPILABLE: LayerCompilable (malop ⊕ L64),
    ∀ MOK: ModuleOK (pfree ↦ f_pfree),
    ∀ M2: LAsm.module,
      CompCertiKOS.transf_module (pfree ↦ f_pfree) = OK M2 →
      cl_sim _ _
             (crel HDATA LDATA)
             (pfree ↦ gensem pfree´_spec)
             (〚 M2 〛(malop ⊕ L64)).
  Proof.
    intros.
    eapply link_compiler; eauto.
    - eapply pfree_spec_ref.
    - link_nextblock.
    - link_kernel_mode.
    - eapply MALOPCODE.pfree_code_correct.

  Qed.
