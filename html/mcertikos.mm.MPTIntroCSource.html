      extern void mem_init(unsigned int);

      void idpde_init(unsigned int mbi_adr)
      {
        unsigned int i, j;
        unsigned int perm;
        mem_init(mbi_adr);
        for(i = 0; i < 1024; i ++)
        {
          if (i < 256)
            perm = PT_PERM_PTKT;
          else if (i >= 960)
            perm = PT_PERM_PTKT;
          else
            perm = PT_PERM_PTKF;
          for(j = 0; j < 1024; j ++)
          {
            set_IDPTE(i, j, perm);
          }
        }
      }
