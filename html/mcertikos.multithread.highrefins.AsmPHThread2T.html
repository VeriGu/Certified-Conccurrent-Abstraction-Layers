    ∀ ef,
      (match ef with
         | EF_external id _ ⇒
           if peq id thread_yield then False
           else if peq id thread_sleep then False
                else if has_event id then True else False

       end)

          | EF_external id _ ⇒

          end)

            | EF_external id _ ⇒

            | EF_external id _ ⇒

              else False
