  Lemma val_inject_vint_eq:
     f n v v´ i,
      val_inject f (ZMap.get n v) (ZMap.get n v´)
       ZMap.get n v = Vint i
       ZMap.get n v´ = Vint i.
  Proof.
    intros.
    rewrite H0 in H; inversion H; reflexivity.
  Qed.
