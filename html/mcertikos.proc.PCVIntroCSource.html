    idq_push(unsigned int chid, unsigned int qid, unsigned int tid)
    {

      if (h == num_proc)
      {
        fifobbq_set_head(chid, qid, tid);
        fifobbq_set_tail(chid, qid, tid);
      }
      else
      {
        t = fifobbq_get_tail(chid, qid);
        fifobbq_set_head(chid, qid, h);
        fifobbq_set_tail(chid, qid, tid);
        idq_set_next(t, tid);
        idq_set_prev(tid, t);
        idq_set_next(tid, num_proc);
      }
    }
