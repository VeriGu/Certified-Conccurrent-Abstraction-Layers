      void ept_init(void)
      {
        unsigned int i, j;

        set_EPML4E(0);
        i = 0;
        while(i < 4)
        {
          set_EPDPTE(0, i);
          j = 0;
          while (j < 512)
          {
            set_EPDTE(0, i, j);
            j ++;
          }
          i ++;
        }
      }
