              ∀ n (d: RData),
              ∃ d´, (d {multi_log : real_multi_log_pb n (multi_log d)}) = d´.
            Proof.
              clearAll.
              induction n; intros.
              ∃ (d {multi_log
                         : real_multi_log_pb 0 (multi_log d)}); auto.
              destruct IHn with d as (d´, H).
              simpl.
              ∃
                d {multi_log
                   : ZMap.set (Z.of_nat n) (MultiDef nil)
                              (multi_log d´)}.
              Opaque real_multi_log_pb.
              clear IHn. destruct d; destruct d´; simpl in ×.
              inv H; simpl. repeat f_equal.
              Transparent real_multi_log_pb.
